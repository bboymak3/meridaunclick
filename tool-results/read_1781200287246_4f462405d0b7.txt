   123→    // ─── Tab Navigation ─────────────────────────────────────────
   124→    function setupTabNavigation() {
   125→        document.querySelectorAll('.admin-nav-link').forEach(link => {
   126→            link.addEventListener('click', (e) => {
   127→                e.preventDefault();
   128→                const tab = link.dataset.tab;
   129→                if (tab) switchTab(tab);
   130→            });
   131→        });
   132→
   133→        // Also handle data-tab links inside cards
   134→        document.querySelectorAll('[data-tab]').forEach(el => {
   135→            if (el.tagName === 'A') {
   136→                el.addEventListener('click', (e) => {
   137→                    e.preventDefault();
   138→                    const tab = el.dataset.tab;
   139→                    if (tab) switchTab(tab);
   140→                });
   141→            }
   142→        });
   143→    }

   145→    function switchTab(tab) {
   146→        currentTab = tab;
   147→
   148→        // Update nav links
   149→        document.querySelectorAll('.admin-nav-link').forEach(link => {
   150→            link.classList.toggle('active', link.dataset.tab === tab);
   151→        });
   152→
   153→        // Hide all tab panels
   154→        const panels = { dashboard: tabDashboard, businesses: tabProperties, users: tabUsers, messages: tabMessages, facebook: tabFacebook, jobs: tabJobs, plans: tabPlans, settings: tabSettings };
   155→        for (const [key, panel] of Object.entries(panels)) {
   156→            if (panel) {
   157→                panel.classList.toggle('hidden', key !== tab);
   158→            }
   159→        }
   160→
   161→        // Update page title
   162→        const titles = { dashboard: 'Dashboard', businesses: 'Negocios', users: 'Usuarios', messages: 'Mensajes', facebook: 'Facebook Import', jobs: 'Empleo', plans: 'Planes', settings: 'Configuración' };
   163→        if (adminPageTitle) {
   164→            adminPageTitle.textContent = titles[tab] || 'Dashboard';
   165→        }
   166→
   167→        // Load tab data
   168→        switch (tab) {
   169→            case 'dashboard':
   170→                loadDashboardTab();
   171→                break;
   172→            case 'businesses':
   173→                propsPage = 1;
   174→                loadProperties();
   175→                break;
   176→            case 'users':
   177→                usersPage = 1;
   178→                loadUsers();
   179→                break;
   180→            case 'messages':
   181→                msgsPage = 1;
   182→                loadMessages();
   183→                break;
   184→            case 'facebook':
   185→                loadFacebookConfig();
   186→                loadFacebookHistory();
   187→                break;
   188→            case 'products':
   189→                productsPage = 1;
   190→                loadProducts();
   191→                break;
   192→            case 'jobs':
   193→                jobsPage = 1;
   194→                loadJobs();
   195→                break;
   196→            case 'properties':
   197→                propertiesPage = 1;
   198→                loadProperties();
   199→                break;
   200→            case 'plans':
   201→                loadPlansTab();
   202→                break;
   203→            case 'settings':
   204→                loadSettings();
   205→                break;
   206→        }
   207→
   208→        // Close sidebar on mobile
   209→        if (adminSidebar) {
   210→            adminSidebar.classList.remove('active');
   211→        }
   212→    }

   946→    // ─── Image Compression (client-side) ────────────────────────
   947→    // Reduces image resolution before upload to save R2 storage and bandwidth
   948→    // Max longest side: 1920px, JPEG/WebP quality: 0.75
   949→    function compressImage(file, maxSize, quality) {
   950→        maxSize = maxSize || 1920;
   951→        quality = quality || 0.75;
   952→        return new Promise((resolve, reject) => {
   953→            // Don't compress GIFs or tiny files (< 200KB)
   954→            if (file.type === 'image/gif' || file.size < 200 * 1024) {
   955→                resolve(file);
   956→                return;
   957→            }
   958→            const img = new Image();
   959→            const url = URL.createObjectURL(file);
   960→            img.onload = function () {
   961→                URL.revokeObjectURL(url);
   962→                let w = img.width, h = img.height;
   963→                // Only resize if exceeds max dimension
   964→                if (w <= maxSize && h <= maxSize) {
   965→                    resolve(file);
   966→                    return;
   967→                }
   968→                if (w > h) { h = Math.round(h * maxSize / w); w = maxSize; }
   969→                else { w = Math.round(w * maxSize / h); h = maxSize; }
   970→                const canvas = document.createElement('canvas');
   971→                canvas.width = w;
   972→                canvas.height = h;
   973→                const ctx = canvas.getContext('2d');
   974→                ctx.drawImage(img, 0, 0, w, h);
   975→                // Use WebP if supported, else JPEG
   976→                const mimeType = canvas.toDataURL('image/webp').startsWith('data:image/webp') ? 'image/webp' : 'image/jpeg';
   977→                const ext = mimeType === 'image/webp' ? 'webp' : 'jpg';
   978→                canvas.toBlob(function (blob) {
   979→                    if (!blob) { resolve(file); return; }
   980→                    const compressed = new File([blob], file.name.replace(/\.[^.]+$/, '.') + ext, { type: mimeType });
   981→                    const pct = Math.round((1 - compressed.size / file.size) * 100);
   982→                    console.log(`Image compressed: ${(file.size/1024/1024).toFixed(1)}MB → ${(compressed.size/1024/1024).toFixed(1)}MB (${pct}% reduction)`);
   983→                    resolve(compressed);
   984→                }, mimeType, quality);
   985→            };
   986→            img.onerror = function () { URL.revokeObjectURL(url); resolve(file); };
   987→            img.src = url;
   988→        });
   989→    }

  1261→    function renderAdminPagination(containerId, pagination, onPageChange) {

  1263→        if (!container || !pagination) return;

  1271→
  1272→        let pagesHTML = '';
  1273→

  1276→

  1278→        const maxVisible = 5;
  1279→        let startPage = Math.max(1, page - Math.floor(maxVisible / 2));
  1280→        let endPage = Math.min(totalPages, startPage + maxVisible - 1);
  1281→        if (endPage - startPage < maxVisible - 1) {
  1282→            startPage = Math.max(1, endPage - maxVisible + 1);
  1283→        }
  1284→
  1285→        if (startPage > 1) {
  1286→            pagesHTML += `<button class="pagination-btn" data-page="1">1</button>`;
  1287→            if (startPage > 2) pagesHTML += '<span class="pagination-dots">...</span>';
  1288→        }
  1289→
  1290→        for (let i = startPage; i <= endPage; i++) {
  1291→            pagesHTML += `<button class="pagination-btn ${i === page ? 'active' : ''}" data-page="${i}">${i}</button>`;
  1292→        }
  1293→
  1294→        if (endPage < totalPages) {
  1295→            if (endPage < totalPages - 1) pagesHTML += '<span class="pagination-dots">...</span>';
  1296→            pagesHTML += `<button class="pagination-btn" data-page="${totalPages}">${totalPages}</button>`;
  1297→        }

  1299→        // Next button

  1301→
  1302→        // Total info
  1303→        pagesHTML += `<span class="pagination-info">Total: ${total}</span>`;
  1304→
  1305→        container.innerHTML = pagesHTML;

  1307→        // Click handlers
  1308→        container.querySelectorAll('.pagination-btn:not([disabled])').forEach(btn => {
  1309→            btn.addEventListener('click', () => {
  1310→                const p = parseInt(btn.dataset.page);
  1311→                if (p && p >= 1 && p <= totalPages) {
