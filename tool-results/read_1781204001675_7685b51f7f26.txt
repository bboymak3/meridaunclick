   119→    // ─── Tab Navigation ─────────────────────────────────────────
   120→    function setupTabNavigation() {
   121→        document.querySelectorAll('.admin-nav-link').forEach(link => {
   122→            link.addEventListener('click', (e) => {
   123→                e.preventDefault();
   124→                const tab = link.dataset.tab;
   125→                if (tab) switchTab(tab);
   126→            });
   127→        });
   128→
   129→        // Also handle data-tab links inside cards
   130→        document.querySelectorAll('[data-tab]').forEach(el => {
   131→            if (el.tagName === 'A') {
   132→                el.addEventListener('click', (e) => {
   133→                    e.preventDefault();
   134→                    const tab = el.dataset.tab;
   135→                    if (tab) switchTab(tab);
   136→                });
   137→            }
   138→        });
   139→    }

   141→    function switchTab(tab) {
   142→        currentTab = tab;
   143→
   144→        // Update nav links
   145→        document.querySelectorAll('.admin-nav-link').forEach(link => {
   146→            link.classList.toggle('active', link.dataset.tab === tab);
   147→        });
   148→
   149→        // Hide all tab panels
   150→        const panels = { dashboard: tabDashboard, businesses: tabProperties, users: tabUsers, messages: tabMessages, facebook: tabFacebook, jobs: tabJobs, settings: tabSettings };
   151→        for (const [key, panel] of Object.entries(panels)) {
   152→            if (panel) {
   153→                panel.classList.toggle('hidden', key !== tab);
   154→            }
   155→        }
   156→
   157→        // Update page title
   158→        const titles = { dashboard: 'Dashboard', businesses: 'Negocios', users: 'Usuarios', messages: 'Mensajes', facebook: 'Facebook Import', jobs: 'Empleo', settings: 'Configuración' };
   159→        if (adminPageTitle) {
   160→            adminPageTitle.textContent = titles[tab] || 'Dashboard';
   161→        }
   162→
   163→        // Load tab data
   164→        switch (tab) {
   165→            case 'dashboard':
   166→                loadDashboardTab();
   167→                break;
   168→            case 'businesses':
   169→                propsPage = 1;
   170→                loadProperties();
   171→                break;
   172→            case 'users':
   173→                usersPage = 1;
   174→                loadUsers();
   175→                break;
   176→            case 'messages':
   177→                msgsPage = 1;
   178→                loadMessages();
   179→                break;
   180→            case 'facebook':
   181→                loadFacebookConfig();
   182→                loadFacebookHistory();
   183→                break;
   184→            case 'products':
   185→                productsPage = 1;
   186→                loadProducts();
   187→                break;
   188→            case 'jobs':
   189→                jobsPage = 1;
   190→                loadJobs();
   191→                break;
   192→            case 'settings':
   193→                loadSettings();
   194→                break;
   195→        }
   196→
   197→        // Close sidebar on mobile
   198→        if (adminSidebar) {
   199→            adminSidebar.classList.remove('active');
   200→        }
   201→    }

   899→    // ─── Image Compression (client-side) ────────────────────────
   900→    // Reduces image resolution before upload to save R2 storage and bandwidth
   901→    // Max longest side: 1920px, JPEG/WebP quality: 0.75
   902→    function compressImage(file, maxSize, quality) {
   903→        maxSize = maxSize || 1920;
   904→        quality = quality || 0.75;
   905→        return new Promise((resolve, reject) => {
   906→            // Don't compress GIFs or tiny files (< 200KB)
   907→            if (file.type === 'image/gif' || file.size < 200 * 1024) {
   908→                resolve(file);
   909→                return;
   910→            }
   911→            const img = new Image();
   912→            const url = URL.createObjectURL(file);
   913→            img.onload = function () {
   914→                URL.revokeObjectURL(url);
   915→                let w = img.width, h = img.height;
   916→                // Only resize if exceeds max dimension
   917→                if (w <= maxSize && h <= maxSize) {
   918→                    resolve(file);
   919→                    return;
   920→                }
   921→                if (w > h) { h = Math.round(h * maxSize / w); w = maxSize; }
   922→                else { w = Math.round(w * maxSize / h); h = maxSize; }
   923→                const canvas = document.createElement('canvas');
   924→                canvas.width = w;
   925→                canvas.height = h;
   926→                const ctx = canvas.getContext('2d');
   927→                ctx.drawImage(img, 0, 0, w, h);
   928→                // Use WebP if supported, else JPEG
   929→                const mimeType = canvas.toDataURL('image/webp').startsWith('data:image/webp') ? 'image/webp' : 'image/jpeg';
   930→                const ext = mimeType === 'image/webp' ? 'webp' : 'jpg';
   931→                canvas.toBlob(function (blob) {
   932→                    if (!blob) { resolve(file); return; }
   933→                    const compressed = new File([blob], file.name.replace(/\.[^.]+$/, '.') + ext, { type: mimeType });
   934→                    const pct = Math.round((1 - compressed.size / file.size) * 100);
   935→                    console.log(`Image compressed: ${(file.size/1024/1024).toFixed(1)}MB → ${(compressed.size/1024/1024).toFixed(1)}MB (${pct}% reduction)`);
   936→                    resolve(compressed);
   937→                }, mimeType, quality);
   938→            };
   939→            img.onerror = function () { URL.revokeObjectURL(url); resolve(file); };
   940→            img.src = url;
   941→        });
   942→    }

   944→    async function handleBizImageUpload(file) {
   945→        if (!file) return;
   946→        const statusEl = document.getElementById('bizEditImageStatus');
   947→        const businessId = adminBizEditModal?.dataset?.businessId;
   948→
   949→        if (!businessId) { showToast('Error: no se identificó el negocio', 'error'); return; }
   950→
   951→        if (statusEl) statusEl.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Subiendo...';
   952→
   953→        try {
   954→            // Step 1: Upload file to R2
   955→            const formData = new FormData();
   956→            formData.append('file', file);
   957→            formData.append('business_id', businessId);
   958→            formData.append('product_type', 'business');
   959→
   960→            const token = localStorage.getItem(TOKEN_KEY);
   961→            const response = await fetch('/api/upload', {

   965→            });
   966→            const data = await response.json();
   967→
   968→            if (!data.url) {
   969→                if (statusEl) statusEl.innerHTML = '<i class="fas fa-exclamation-triangle" style="color:#dc3545;"></i> ' + (data.error || 'Error al subir');

  1207→    // ─── Pagination Helper ──────────────────────────────────────
  1208→    function renderAdminPagination(containerId, pagination, onPageChange) {
  1209→        const container = document.getElementById(containerId);
  1210→        if (!container || !pagination) return;
  1211→
  1212→        const { page, totalPages, total } = pagination;
  1213→
  1214→        if (totalPages <= 1) {
  1215→            container.innerHTML = '';
  1216→            return;
  1217→        }
  1218→
  1219→        let pagesHTML = '';
  1220→
  1221→        // Previous button
  1222→        pagesHTML += `<button class="pagination-btn" ${page <= 1 ? 'disabled' : ''} data-page="${page - 1}"><i class="fas fa-chevron-left"></i></button>`;
  1223→
  1224→        // Page numbers
  1225→        const maxVisible = 5;
  1226→        let startPage = Math.max(1, page - Math.floor(maxVisible / 2));
  1227→        let endPage = Math.min(totalPages, startPage + maxVisible - 1);
  1228→        if (endPage - startPage < maxVisible - 1) {
  1229→            startPage = Math.max(1, endPage - maxVisible + 1);
  1230→        }
  1231→
  1232→        if (startPage > 1) {
  1233→            pagesHTML += `<button class="pagination-btn" data-page="1">1</button>`;
  1234→            if (startPage > 2) pagesHTML += '<span class="pagination-dots">...</span>';
  1235→        }
  1236→
  1237→        for (let i = startPage; i <= endPage; i++) {
  1238→            pagesHTML += `<button class="pagination-btn ${i === page ? 'active' : ''}" data-page="${i}">${i}</button>`;
  1239→        }
  1240→
  1241→        if (endPage < totalPages) {
  1242→            if (endPage < totalPages - 1) pagesHTML += '<span class="pagination-dots">...</span>';
  1243→            pagesHTML += `<button class="pagination-btn" data-page="${totalPages}">${totalPages}</button>`;
  1244→        }
  1245→
  1246→        // Next button
  1247→        pagesHTML += `<button class="pagination-btn" ${page >= totalPages ? 'disabled' : ''} data-page="${page + 1}"><i class="fas fa-chevron-right"></i></button>`;
  1248→
  1249→        // Total info
  1250→        pagesHTML += `<span class="pagination-info">Total: ${total}</span>`;
  1251→
  1252→        container.innerHTML = pagesHTML;
  1253→
  1254→        // Click handlers
  1255→        container.querySelectorAll('.pagination-btn:not([disabled])').forEach(btn => {
  1256→            btn.addEventListener('click', () => {
  1257→                const p = parseInt(btn.dataset.page);
  1258→                if (p && p >= 1 && p <= totalPages) {
  1259→                    onPageChange(p);
  1260→                }
  1261→            });
  1262→        });
  1263→    }
